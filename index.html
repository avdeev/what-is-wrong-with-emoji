*️⃣ Универсальное

      '👩🏿'
      .split('')
      .reduce((result, currentChar) => {
        const code = currentChar.charCodeAt(0).toString(16);
        const code4 = `000${code}`.substr(-4);
        return `${result}\\u${code4};`
      }, ''); // \ud83d\udc69\ud83c\udfff
